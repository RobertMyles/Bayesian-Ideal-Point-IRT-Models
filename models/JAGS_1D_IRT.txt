model{
for(i in 1:N){
  for(j in 1:M){
    y[i,j] ~ dbern(pi[i, j])
    logit(pi[i,j]) <- beta[j]*theta[i] -alpha[i]
    }
    }
    
    # constrain the first two voters:
    theta[1] ~ dnorm(1, 1) T(0, )    # truncated to be positive always
    theta[2] ~ dnorm(-1, 1)T(, 0)    # inverse
    
    for(i in 3:N){
    theta[i] ~ dnorm(0, 1)
    }
    
    for(j in 1:M){
    alpha[j] ~ dnorm(0, 5)  # priors the same as pscl::ideal
    beta[j] ~ dnorm(0, 5)
    }
}
